A site I am building on my freetime. 

<h3>Installation:</h3>

	$git clone https://github.com/rochestb/adventurly.git
	$npm install 
	$npm install grunt -g
	$nodemon install -g
	$grunt --watch
	$cd www/
	$npm install
	$mongod 
	$nodemon app.js
